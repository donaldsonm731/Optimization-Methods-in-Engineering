Problem 6a

clc
clf
clear

[x1,x2] = meshgrid(-10:0.01:10,-10:0.01:10 );
f = (x1-3).^2 + (x2-3).^2;

h = x1 - 3*x2 + 1 == 0;
h1 = x1 - 3*x2 + 1 == 1;
h2 = x1 - 3*x2 + 1 == -1;
g = x1 + x2 -4 <= 0;
g1 = x1 + x2 -4 <= 1;
g2 = x1 + x2 -4 <= -1;

contour(x1,x2,f,100)
hold on
contour(x1, x2,h1, '-r')
contour(x1, x2,h2, '-r')
contour(x1, x2,h, '-r')
contour(x1, x2,g,'-g')
contour(x1, x2,g1,'-g')
contour(x1, x2,g2,'-g')
axis([0 5 0 5]);
title('Problem 6')
legend('obj. function', 'h +1', 'h -1', 'h original', 'g original', 'g +1', 'g -1')
