function [f,g1,g2,g3,g4] = p3(x)
d = x(1);
D = x(2);
N = x(3);

% constants
Q = 2;
P = 10;
g = 386;
rho = 7.38342*10^(-4);
G = 1.15*10^7;

t_a = 80000;
delta = 0.5;
w_0 = 100;
D_0 = 1.5;

K = (d^4*G)/(8*D^3*N);

k = (4*D - d)/(4*(D-d))+ 0.615*d/D;
t = (8*k*P*D)/(pi*d^3);

w = d/(2*pi*N*D^2)*sqrt(G/(2*rho));


f = (N + Q)*D.*d.^2;

g1 = -P/K + delta; % Deflection limit
g2 = t - t_a; % Shear Stress
g3 = w_0 - w ; % Freq. surge waves
g4 = D + d - D_0; % Outer Diameter of spring

end
